```tsx
// components/layout/Sidebar.tsx (updated with responsive design)
import { FC } from 'react';
import { useUIStore } from '../../store/uiStore';
import { Button } from '@ui/components/ui/button';
import { ChevronLeftIcon, ChevronRightIcon } from 'lucide-react';

const Sidebar: FC = () => {
  const { sidebarCollapsed, setSidebarCollapsed } = useUIStore();
  
  return (
    <aside 
      className={`
        border-r bg-gray-50 flex flex-col transition-all duration-300
        ${sidebarCollapsed ? 'w-14' : 'w-64 md:w-72'}
      `}
    >
      <div className="flex justify-end p-2">
        <Button 
          variant="ghost" 
          size="sm" 
          onClick={() => setSidebarCollapsed(!sidebarCollapsed)}
          aria-label={sidebarCollapsed ? 'Expand sidebar' : 'Collapse sidebar'}
        >
          {sidebarCollapsed ? <ChevronRightIcon size={16} /> : <ChevronLeftIcon size={16} />}
        </Button>
      </div>
      
      {/* Sidebar content */}
    </aside>
  );
};
```
