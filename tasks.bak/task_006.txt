```tsx
// components/boards/CreateBoardModal.tsx
import { FC, useState } from 'react';
import { useMutation } from 'convex/react';
import { api } from '../../convex/_generated/api';
import { Dialog, DialogContent, DialogHeader, DialogTitle, DialogFooter } from '@ui/components/ui/dialog';
import { Button } from '@ui/components/ui/button';
import { Input } from '@ui/components/ui/input';

interface CreateBoardModalProps {
  isOpen: boolean;
  onClose: () => void;
  folderId?: string; // Optional, if creating within a folder
}

const CreateBoardModal: FC<CreateBoardModalProps> = ({ isOpen, onClose, folderId }) => {
  const [boardName, setBoardName] = useState('New Board');
  const createBoard = useMutation(api.boards.create);
  
  const handleCreate = async () => {
    // Create board with default columns, groups, and items
    await createBoard({
      name: boardName,
      folderId,
      defaultColumns: true, // This will trigger creation of default columns in the mutation
      defaultGroups: true, // This will trigger creation of default groups and items
    });
    onClose();
  };
  
  return (
    <Dialog open={isOpen} onOpenChange={onClose}>
      <DialogContent>
        <DialogHeader>
          <DialogTitle>Create New Board</DialogTitle>
        </DialogHeader>
        <div className="py-4">
          <Input 
            value={boardName} 
            onChange={(e) => setBoardName(e.target.value)} 
            placeholder="Board Name"
            autoFocus
          />
        </div>
        <DialogFooter>
          <Button variant="outline" onClick={onClose}>Cancel</Button>
          <Button onClick={handleCreate}>Create Board</Button>
        </DialogFooter>
      </DialogContent>
    </Dialog>
  );
};

export default CreateBoardModal;
```
