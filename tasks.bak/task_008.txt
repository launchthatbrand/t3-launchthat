```tsx
// components/boards/BoardView.tsx
import { FC } from 'react';
import { useQuery } from 'convex/react';
import { api } from '../../convex/_generated/api';
import { Board, Group, Item, Column } from '../../types';
import GroupComponent from './GroupComponent';
import BoardHeader from './BoardHeader';

interface BoardViewProps {
  boardId: string;
}

const BoardView: FC<BoardViewProps> = ({ boardId }) => {
  const board = useQuery(api.boards.getById, { id: boardId });
  const groups = useQuery(api.groups.getByBoardId, { boardId }) || [];
  
  if (!board) return <div>Loading board...</div>;
  
  return (
    <div className="flex flex-col h-full">
      <BoardHeader board={board} />
      
      <div className="flex-1 overflow-auto">
        {groups.map(group => (
          <GroupComponent 
            key={group.id} 
            group={group} 
            columns={board.columns} 
            boardId={boardId} 
          />
        ))}
      </div>
    </div>
  );
};

export default BoardView;
```

Implement the supporting components:
- `BoardHeader.tsx` for displaying and managing columns
- `GroupComponent.tsx` for rendering a group and its items
- `ItemRow.tsx` for rendering an individual item as a row
- Cell components for different column types (text, date, dropdown, etc.)
