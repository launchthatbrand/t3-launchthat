```tsx
// components/columns/ItemCell.tsx
import { FC, useState } from 'react';
import { Input } from '@ui/components/ui/input';
import { Textarea } from '@ui/components/ui/textarea';
import { Checkbox } from '@ui/components/ui/checkbox';
import { Calendar } from '@ui/components/ui/calendar';
import { Select, SelectContent, SelectItem, SelectTrigger, SelectValue } from '@ui/components/ui/select';
import { Popover, PopoverContent, PopoverTrigger } from '@ui/components/ui/popover';
import { Button } from '@ui/components/ui/button';
import { CalendarIcon } from 'lucide-react';
import { format } from 'date-fns';
import { ColumnType } from '../../types';

interface ItemCellProps {
  type: ColumnType;
  value: any;
  onChange: (value: any) => void;
  options?: string[]; // For dropdown type
}

const ItemCell: FC<ItemCellProps> = ({ type, value, onChange, options = [] }) => {
  const [isEditing, setIsEditing] = useState(false);
  
  // Helper to handle blur events
  const handleBlur = () => setIsEditing(false);
  
  switch (type) {
    case 'text':
      return isEditing ? (
        <Input 
          value={value || ''}
          onChange={(e) => onChange(e.target.value)}
          onBlur={handleBlur}
          autoFocus
        />
      ) : (
        <div onClick={() => setIsEditing(true)} className="cursor-text min-h-[24px]">
          {value || ''}
        </div>
      );
      
    case 'longText':
      return isEditing ? (
        <Textarea 
          value={value || ''}
          onChange={(e) => onChange(e.target.value)}
          onBlur={handleBlur}
          autoFocus
        />
      ) : (
        <div onClick={() => setIsEditing(true)} className="cursor-text min-h-[24px]">
          {value || ''}
        </div>
      );
    
    // Implement other column types similarly
    
    default:
      return <div>Unsupported column type: {type}</div>;
  }
};

export default ItemCell;
```

Implement the remaining column type components following the same pattern, using appropriate Shadcn UI components for each type.
